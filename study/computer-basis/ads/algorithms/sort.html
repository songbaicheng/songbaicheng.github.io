              public void insertSort(E[] arr) {
    for (int i = 1; i < arr.length; i++) {

        E tempElement = arr[i];
        int j = i - 1;

        while (j >= 0 && tempElement.compareTo(arr[j]) < 0) {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = tempElement;
    }
